Build + sign + verify (one command)

export COSIGN_PASSWORD=""
COSIGN_KEY=.secret/cosign_signing_key.key COSIGN_PUB=.secret/cosign_signing_key.pub \
  bash build_sign_all.sh | tee BUILD_RUN.log
# -> artifacts/zarf-package-*-0.3.1.tar.zst + *.uds.sig + SBOM + SHA256SUMS.txt
# -> cosign verify-blob: Verified OK ; zarf package inspect: component/image listing

Deploy on UDS Core

uds deploy k3d-core-slim-dev:0.41.0 --confirm
for f in a11oy amaru sentra killinchu rosie; do
  ( cd PER_BUNDLE/$f && uds create . --confirm && \
    uds deploy uds-bundle-$f-amd64-0.3.1.tar.zst --confirm )
done
