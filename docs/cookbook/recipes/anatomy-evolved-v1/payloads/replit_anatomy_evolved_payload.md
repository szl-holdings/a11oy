## 2.3 TypeScript — DARUAN Single-Qubit Activation

```typescript
// DARUAN: DatA Re-Uploading ActivatioN
// Single-qubit data re-uploading circuit as learnable nonlinear activation
// Source: arXiv:2605.06734 (Peng et al., 2026)

export interface DaruanParams {
  theta: number[]; // rotation angles [θ₁, θ₂, θ₃]
  phi: number[];   // phase angles   [φ₁, φ₂, φ₃]
  layers: number;  // re-uploading layers L (default 3)
}

export interface QubitState {
  alpha: [number, number]; // complex amplitude |0⟩
  beta:  [number, number]; // complex amplitude |1⟩
}

/**
 * DARUAN activation: L layers of Rz(φ·x) Rx(θ) data re-uploading
 * Correct real-arithmetic single-qubit circuit simulation.
 * Output: expectation value ⟨Z⟩ = |α|² - |β|² ∈ [-1, +1]
 * State: [re(α), im(α), re(β), im(β)] for |ψ⟩ = α|0⟩ + β|1⟩
 * Rz(φ): α *= e^{-iφ/2}, β *= e^{+iφ/2}
 * Rx(θ): [[cos(θ/2), -i·sin(θ/2)], [-i·sin(θ/2), cos(θ/2)]]
 */
export function daruanActivate(x: number, params: DaruanParams): number {
  let re_a = 1, im_a = 0, re_b = 0, im_b = 0; // start |0⟩
  for (let l = 0; l < params.layers; l++) {
    const idx = l % params.theta.length;
    // Rz(phi·x): alpha *= e^{-i·phi·x/2}, beta *= e^{+i·phi·x/2}
    const az = params.phi[idx] * x / 2;
    const cz = Math.cos(az), sz = Math.sin(az);
    const [ra, ia, rb, ib] = [
      re_a * cz + im_a * sz, im_a * cz - re_a * sz,
      re_b * cz - im_b * sz, im_b * cz + re_b * sz,
    ];
    // Rx(theta): [[cos(t/2), -i·sin(t/2)], [-i·sin(t/2), cos(t/2)]]
    const t2 = params.theta[idx] / 2;
    const ct = Math.cos(t2), st = Math.sin(t2);
    [re_a, im_a, re_b, im_b] = [
      ct * ra + st * ib,  ct * ia - st * rb,
      ct * rb + st * ia,  ct * ib - st * ra,
    ];
  }
  return (re_a * re_a + im_a * im_a) - (re_b * re_b + im_b * im_b); // ⟨Z⟩
}
```

## 2.4 TypeScript — Scalar-Gated Fast-Weight Update

```typescript
// Scalar-gated fast-weight programmer update rule
// Source: arXiv:2605.06734 §3 — geometric boundedness by construction

export interface FastWeightMatrix {
  W: Float64Array; // flattened d×d matrix
  d: number;
}

/** Scalar gate σ(g) ∈ (0, 1) — bounds the update magnitude */
function scalarGate(g: number): number {
  return 1.0 / (1.0 + Math.exp(-g)); // sigmoid gate
}

/**
 * Gated fast-weight update:
 *   W_{t+1} = (1 - σ(g)) · W_t + σ(g) · outer(k, v)
 * Geometric boundedness: ||W_t||_F ≤ B for all t
 * (proved in lutar-lean/Lutar/QKAN/GatedBoundedness.lean)
 */
export function gatedFastWeightUpdate(
  W: FastWeightMatrix,
  k: Float64Array,
  v: Float64Array,
  g: number,
): FastWeightMatrix {
  const gate = scalarGate(g);
  const decay = 1.0 - gate;
  const { d } = W;
  const newW = new Float64Array(d * d);
  for (let i = 0; i < d; i++) {
    for (let j = 0; j < d; j++) {
      newW[i * d + j] = decay * W.W[i * d + j] + gate * k[i] * v[j];
    }
  }
  return { W: newW, d };
}

/** Frobenius norm — used to verify geometric boundedness invariant */
export function frobeniusNorm(W: FastWeightMatrix): number {
  let sum = 0;
  for (const w of W.W) sum += w * w;
  return Math.sqrt(sum);
}
```

## 2.5 Composition with Bohr Complementarity

The scalar gate value encodes the frame weight ratio for each complementary pair:

```typescript
// gate = 0.0 → pure "intent frame" DARUAN
// gate = 0.5 → equal weight (maximum complementarity uncertainty)
// gate = 1.0 → pure "effect frame" DARUAN
// Invariant: σA · σB ≥ COMPLEMENTARITY_FLOOR (Bohr uncertainty relation)
// where σA = gate, σB = 1 - gate, FLOOR = 0.25 (at 50/50 split)

export function complementarityGate(intentScore: number, effectScore: number): number {
  // Returns gate value that minimizes weighted loss while respecting uncertainty floor
  const FLOOR = 0.25;
  const rawGate = intentScore / (intentScore + effectScore + 1e-8);
  // Clamp to [FLOOR, 1-FLOOR] so complementarity floor is never violated
  return Math.max(FLOOR, Math.min(1 - FLOOR, rawGate));
}
```

```lean
-- lutar-lean/Lutar/QKAN/GatedBoundedness.lean
-- Source: arXiv:2605.06734 — scalar-gated fast-weight update
-- Author: Stephen P. Lutar Jr., SZL Holdings

import Mathlib.Analysis.NormedSpace.Basic
import Mathlib.Topology.Algebra.Module.FiniteDimension

-- Gate function: σ(g) ∈ (0, 1)
noncomputable def scalarGate (g : ℝ) : ℝ := 1 / (1 + Real.exp (-g))

-- Gate is in (0, 1) for all real g
lemma scalarGate_pos (g : ℝ) : 0 < scalarGate g := by
  unfold scalarGate
  positivity

lemma scalarGate_lt_one (g : ℝ) : scalarGate g < 1 := by
  unfold scalarGate
  have h : 0 < Real.exp (-g) := Real.exp_pos _
  linarith [h]

-- Fast-weight update: W_{t+1} = (1 - σ) · W_t + σ · outer(k, v)
-- Boundedness: if ||W_t||_F ≤ B and ||k||·||v|| ≤ B, then ||W_{t+1}||_F ≤ B
theorem gated_qkan_boundedness
    {n : ℕ} (B : ℝ) (hB : 0 < B)
    (W : Matrix (Fin n) (Fin n) ℝ) (k v : Fin n → ℝ)
    (g : ℝ)
    (hW : Matrix.frobeniusNorm W ≤ B)
    (hkv : ‖k‖ * ‖v‖ ≤ B) :
    Matrix.frobeniusNorm
      ((1 - scalarGate g) • W + (scalarGate g) • Matrix.vecMulVec k v)
      ≤ B := by
  have hσ := scalarGate_pos g
  have hσ1 := scalarGate_lt_one g
  -- Triangle inequality on Frobenius norm
  calc Matrix.frobeniusNorm ((1 - scalarGate g) • W + scalarGate g • Matrix.vecMulVec k v)
      ≤ Matrix.frobeniusNorm ((1 - scalarGate g) • W) +
        Matrix.frobeniusNorm (scalarGate g • Matrix.vecMulVec k v) := by
          apply Matrix.frobeniusNorm_add_le
    _ = (1 - scalarGate g) * Matrix.frobeniusNorm W +
        scalarGate g * Matrix.frobeniusNorm (Matrix.vecMulVec k v) := by
          simp [Matrix.frobeniusNorm_smul]
    _ ≤ (1 - scalarGate g) * B + scalarGate g * B := by
          have hvv : Matrix.frobeniusNorm (Matrix.vecMulVec k v) ≤ B := by
            sorry -- ‖outer(k,v)‖_F = ‖k‖·‖v‖ ≤ B (1-line Cauchy-Schwarz)
          linarith [mul_le_mul_of_nonneg_left hW (by linarith),
                    mul_le_mul_of_nonneg_left hvv (le_of_lt hσ)]
    _ = B := by ring
```

# Part 4 — Willow / Post-Quantum Migration Plan

## 4.1 What Willow Actually Proved

Google Willow (Acharya et al., *Nature* 638, 920–926, 2025; arXiv:2408.13687) proved **below-threshold quantum error correction** on a superconducting 105-qubit chip:

- Error suppression factor Λ = 2.14 ± 0.02 per code-distance step
- At d=7: logical error rate 0.143% ± 0.003% per cycle
- Logical qubit lifetime 291 ± 6 µs, exceeding best physical qubit (119 µs) by 2.4×
- Real-time decoder: 63 µs at d=5 (within coherence time)

**This is a genuine scientific milestone.** Below-threshold operation was the central unsolved experimental problem in quantum error correction since Shor 1995.

## 4.2 What Willow Does NOT Prove

1. **Not useful computation yet.** The d=7 memory experiment ran no T-gates. Magic-state distillation (needed for universal computation) was not demonstrated and consumes 90%+ of physical qubits.
2. **Not a cryptographic threat.** The RCS benchmark proves quantum advantage on a task purposely hard for classical computers, not on Shor's algorithm. These are orthogonal claims.
3. **Not scalable without magic-state overhead solved.** Reaching 10⁻¹⁵ logical error rate requires d ≈ 67, meaning ~9,000 physical qubits per logical qubit at current Λ.
4. **Not independently verified.** The 10²⁵-year classical simulation claim cannot be directly verified — it extrapolates from smaller circuits.

**Revised threat timeline (from quantum_leaders_report.md §5):** Preskill/Cain (arXiv:2603.28627, Mar 2026) revised minimum qubit threshold for RSA-2048 factoring to 10,000–20,000 neutral-atom qubits using high-rate LDPC codes — a 1,000× reduction from earlier estimates. The realistic first-mover Shor threat is **2028–2032** (QuEra, IonQ, or neutral-atom-class platform).

```typescript
// ─── DARUAN Core ─────────────────────────────────────────────────────────────

interface QubitState {
  alpha: [number, number]; // [re, im] of |0⟩ amplitude
  beta:  [number, number]; // [re, im] of |1⟩ amplitude
}

function rx(s: QubitState, theta: number): QubitState {
  const c = Math.cos(theta / 2), si = Math.sin(theta / 2);
  return {
    alpha: [c * s.alpha[0] - si * s.beta[1], c * s.alpha[1] + si * s.beta[0]],
    beta:  [si * s.alpha[1] + c * s.beta[0], -si * s.alpha[0] + c * s.beta[1]],
  };
}

function rz(s: QubitState, phi: number): QubitState {
  const cp = Math.cos(phi / 2), sp = Math.sin(phi / 2);
  return {
    alpha: [cp * s.alpha[0] - sp * s.alpha[1], cp * s.alpha[1] + sp * s.alpha[0]],
    beta:  [cp * s.beta[0]  + sp * s.beta[1],  cp * s.beta[1]  - sp * s.beta[0]],
  };
}

/** DARUAN forward pass: L layers of Rz(φ·x) Rx(θ) → expectation ⟨Z⟩ ∈ [-1,+1] */
export function daruanActivate(x: number, params: DaruanParams): number {
  let state: QubitState = { alpha: [1, 0], beta: [0, 0] };
  for (let l = 0; l < params.layers; l++) {
    const idx = l % params.theta.length;
    state = rz(state, params.phi[idx] * x);
    state = rx(state, params.theta[idx]);
  }
  const a2 = state.alpha[0] ** 2 + state.alpha[1] ** 2;
  const b2 = state.beta[0]  ** 2 + state.beta[1]  ** 2;
  return a2 - b2; // ⟨Z⟩
}

// ─── Scalar-Gated Fast-Weight Update ─────────────────────────────────────────

function sigmoid(x: number): number { return 1 / (1 + Math.exp(-x)); }

export function frobeniusNorm(W: FastWeightMatrix): number {
  let sum = 0;
  for (const w of W.W) sum += w * w;
  return Math.sqrt(sum);
}

/**
 * Gated fast-weight update — geometric boundedness guaranteed
 * Lean theorem: gated_qkan_boundedness (lutar-lean/Lutar/QKAN/GatedBoundedness.lean)
 */
export function gatedUpdate(
  W: FastWeightMatrix,
  key: Float64Array,
  value: Float64Array,
  g: number,
): FastWeightMatrix {
  const gate = sigmoid(g);
  const decay = 1 - gate;
  const { d } = W;
  const newW = new Float64Array(d * d);
  for (let i = 0; i < d; i++) {
    for (let j = 0; j < d; j++) {
      newW[i * d + j] = decay * W.W[i * d + j] + gate * key[i] * value[j];
    }
  }
  return { W: newW, d };
}

// ─── QKAN-FWP Sequence Model ──────────────────────────────────────────────────

/** Query the fast-weight memory with key k */
export function fastWeightQuery(W: FastWeightMatrix, k: Float64Array): Float64Array {
  const { d } = W;
  const out = new Float64Array(d);
  for (let i = 0; i < d; i++) {
    let s = 0;
    for (let j = 0; j < d; j++) s += W.W[i * d + j] * k[j];
    out[i] = s;
  }
  return out;
}

  const { fastWeights, gates, params } = state;
  let x = inputVec;
  const nextWeights: FastWeightMatrix[] = [];

  for (let l = 0; l < fastWeights.length; l++) {
    // DARUAN activation on each element of x
```

```typescript
/** Build a canonical 3-outcome policy POVM (accept/abstain/reject) */
export function buildCanonicalPolicyPOVM(alpha: number, beta: number): POVM {
  // E_accept = α·[[1,0],[0,0]], E_reject = β·[[0,0],[0,1]], E_abstain = (1-α-β)·I
  if (alpha + beta > 1) throw new Error("α + β must be ≤ 1");
  const elements: POVMElement[] = [
    {
      id: "ACCEPT", label: "Accept",
      probability: alpha,
      operator: [[alpha, 0], [0, 0]],
    },
    {
      id: "ABSTAIN", label: "Abstain",
      probability: 1 - alpha - beta,
      operator: [[1 - alpha - beta, 0], [0, 1 - alpha - beta]],
    },
    {
      id: "REJECT", label: "Reject",
      probability: beta,
      operator: [[0, 0], [0, beta]],
    },
  ];
  return { elements, isComplete: verifyPOVMCompleteness(elements) };
}

/** Verify completeness: Σ_m E_m = I */
export function verifyPOVMCompleteness(elements: POVMElement[]): boolean {
  const n = 2;
  const sum = Array.from({ length: n }, () => Array(n).fill(0));
  for (const el of elements) {
    for (let i = 0; i < n; i++) {
      for (let j = 0; j < n; j++) {
        sum[i][j] += el.operator[i][j];
      }
    }
  }
  const tol = 1e-9;
  return Math.abs(sum[0][0] - 1) < tol && Math.abs(sum[1][1] - 1) < tol &&
         Math.abs(sum[0][1]) < tol && Math.abs(sum[1][0]) < tol;
}

/** Apply POVM to a density matrix ρ and return verdicts */
export function applyPOVM(povm: POVM, rho: number[][]): POVMVerdict[] {
  return povm.elements.map(el => {
    // p(m) = tr(ρ E_m) = Σ_ij ρ_ij (E_m)_ji
    let p = 0;
    for (let i = 0; i < 2; i++) {
      for (let j = 0; j < 2; j++) {
        p += rho[i][j] * el.operator[j][i];
      }
    }
    return { outcome: el.id, probability: Math.max(0, Math.min(1, p)), confidence: p };
  });
}
```

```typescript
/** Case-insensitive scan of any string for banned tokens */
export function scanForBannedTokens(text: string): DoctrineScanResult {
  const violations: DoctrineScanResult["violations"] = [];
  const lower = text.toLowerCase();

    const t = token.toLowerCase();

      positions.push(pos);
      pos += t.length;

    if (positions.length > 0) violations.push({ token, positions });

}
```

```lean
noncomputable def σ (g : ℝ) : ℝ := 1 / (1 + Real.exp (-g))

lemma σ_lt_one (g : ℝ) : σ g < 1 := by
  unfold σ
  have h : 0 < Real.exp (-g) := Real.exp_pos _
  field_simp; linarith

  unfold σ; field_simp [Real.exp_neg]

-- where k ⊗ v is the outer product (rank-1 matrix)

    {B : ℝ} (hB : 0 < B)
```
